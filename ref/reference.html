Veri Yapıları

cell_asic_

LTC6811 cihazının tüm verilerini içeren ana yapı.

typedef struct {
    uint8_t ic_address;
    cofa_ tx_cfga;
    cofa_ rx_cfga;
    cv_ cell;
    aux_ aux;
    sta_ stata;
    stb_ statb;
    sctrl_ sctrl;
    pwm_ pwm;
    ic_register_ ic_register;
} cell_asic_;

cv_

Hücre voltaj verilerini tutar.

typedef struct {
    uint16_t c_code[12];
} cv_;

aux_

GPIO ve referans voltaj verilerini tutar.

typedef struct {
    uint16_t a_code[5];
    uint16_t ref;
} aux_;

sta_

Status A register verilerini tutar.

typedef struct {
    uint16_t sc;
    uint16_t itmp;
    uint16_t va;
} sta_;

stb_

Status B register verilerini tutar.

typedef struct {
    uint16_t vd;
    uint8_t c_ov[12];
    uint8_t c_uv[12];
    uint8_t thsd : 1;
    uint8_t muxfail : 1;
    uint8_t rsvd : 2;
    uint8_t rev : 4;
} stb_;
